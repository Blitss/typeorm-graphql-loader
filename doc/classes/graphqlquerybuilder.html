loadPaginated

  • loadPaginated<T>(): Promise<[T[], number]>
    • Defined in src/GraphQLQueryBuilder.ts:328

    Loads a paginated set of records via offset and limit (see GraphQLQueryBuilder.paginate) Will return the set of records and overall record count (ignores limit). This count can be used to build the next offset. Please note that the loader will not do any offset/limit calculations for you, it will only apply the values you give it. See Pagination Advice for more info

orWhere

  • orWhere(where: WhereArgument, params?: ObjectLiteral): GraphQLQueryBuilder
    • Defined in src/GraphQLQueryBuilder.ts:145

    Provide the query builder with an OR WHERE condition. Multiple conditions can be added by re-invoking the method (they get added to a list). Any where conditions added via this method will be grouped in an OR expression. Should only be used after an initial WHERE condition.

    example
    // using object literal syntax
function resolve(obj, args, context, info) {
  return context
   .loader
   .loadEntity(Book)
   .info(info)
   .where({authorId: args.authorId})
   .orWhere({isPublicDomain: args.publicDomain})
   .loadMany()
}
    example
    // using custom SQL
 loader
   .loadEntity(Book)
   .info(info)
   .where({authorId: args.authorId})
   // You can use strings with bound values to unlock the full power of SQL
   .orWhere("book.isPublicDomain = :includePublicDomain", {includePublicDomain: args.publicDomain})
   .loadMany()

    Parameters

    • where: WhereArgument

      the WhereArgument you would like applied to the query

    • Optional params: ObjectLiteral

      An optional parameter you can use to bind values for your where condition. See TypeORM docs

    Returns GraphQLQueryBuilder

paginate

  • paginate(pagination: QueryPagination): GraphQLQueryBuilder
    • Defined in src/GraphQLQueryBuilder.ts:269

    Allows you to paginate the query using offset and limit. If used, should be paired with {@link GraphQLQueryBuilder.loadPaginated | loadPaginated}, the other loaders will ignore these options.

    example
    function resolver(obj, args, context, info) {
  const searchOptions: SearchOptions = {
    searchColumns: ["email"],
    searchText: args.searchText
  }

  return context
    .loader
    .loadEntity(User)
    .info(info)
    .search(searchOptions)
    .paginate({offset: args.offset, limit: args.limit})
    .loadManyPaginated()
}

search

  • search(searchOptions: SearchOptions): GraphQLQueryBuilder
    • Defined in src/GraphQLQueryBuilder.ts:190

    Provides an easy way to perform searches without needing to manually write SQL WHERE conditions for each search instance

    The method accepts a searchOptions parameter that allows you to define which columns should be searched, the type of search, and other SQL options.

    example
    function resolver(obj, args, context, info) {
  const searchOptions: SearchOptions = {
    searchColumns: ['email']
    searchText: args.searchText
    // optional
    searchMethod: LoaderSearchMethod.STARTS_WITH
    // optional
    caseSensitive: false
  }

  return context
    .loader
    .loadEntity(User)
    .info(info)
    .search(searchOptions)
    .loadMany()
}

    Parameters

    • searchOptions: SearchOptions

      An SearchOptionsthat allows you to specify which columns should be searched and what to search for

selectFields

  • selectFields(fields: string | Array<string>): GraphQLQueryBuilder
    • Defined in src/GraphQLQueryBuilder.ts:240

    Manually specify fields you always want to be selected. This is useful if you have custom resolvers for other fields on you GraphQL type that may require data that isn't guaranteed to be in a query.

    example
    function resolver(obj, args, context, info) {
  return context
    .loader
    .loadEntity(User)
    .info(info)
    .where({id: args.id})
    // include the email and firstName fields
    // regardless of whether or not they were included in the query
    .selectFields(['email', 'firstName'])
    .loadOne()
}
